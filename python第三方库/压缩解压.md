## 压缩解压

- #### 压缩

  ```python
  import os
  import zipfile
  
  
  def get_zip():
      """
      将本地文件夹进行压缩成zip文件，指定目标路径，名字
      """
      dir_crc = r'F:\图片'      # 被压缩路径
      out_path = r'F:'        # 输出路径
      zip_name = r'图片1234567.zip'     # 文件名
      gen_path = os.path.split(dir_crc)[-1]
      z = zipfile.ZipFile(os.path.join(out_path, zip_name), 'w', zipfile.ZIP_DEFLATED)
      print(os.path.join(out_path, zip_name))     # F:图片1234567.zip
      for dirpath, dirnames, filenames in os.walk(dir_crc):
          fpath = gen_path + dirpath.replace(dir_crc, '') + os.sep
          for filename in filenames:
              print(os.path.join(dirpath, filename), fpath + filename)    # F:\图片\1.jpg 图片\1.jpg
              z.write(os.path.join(dirpath, filename), fpath + filename)
      z.close()
      print('==压缩成功==')
      return ''
  ```
